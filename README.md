## for me

```bash
# from mp3 to mov
for file in audio/*.mp3; do
  filename=$(basename "$file" .mp3)
  ffmpeg -i "$file" -c:a aac -b:a 96k -movflags +faststart "video/${filename}.mov"
done

```
